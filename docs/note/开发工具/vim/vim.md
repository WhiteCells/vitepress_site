```sh
######## 普通模式 #########
[esc]     # 进入普通模式
[u]       # undo，撤销
[ctrl][r] # 恢复上一步撤销的操作

h         # 向左移动
j         # 向下移动
k         # 向上移动
l         # 向右移动

o         # 插入一行
dd        # 删除当前行

[shift][4/$]    # 跳转该行行尾
[shift][6/^]    # 跳转该行行首
[g][g]          # 文本开头
[shift][g]      # 文本末端
{n}[shift][g]   # 跳转到 n 行开头
[w]             # 跳转到下一个单词开头，针对英文
[e]             # 跳转到下一个单词字尾
[b]             # 跳转到上一个单词开头

######## 插入模式 #########
i # insert，插入模式

######## 底行模式 #########
[shift][:] # 低行模式
q # 退出
q! # 未保存无视错误退出
w # 保存
set nu # 显示文件行号
```
